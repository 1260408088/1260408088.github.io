# 枚举简单使用

## 一

``` java
public enum EShare {
    weixing(0), // 微信
    google(1), // 谷歌
    twitter(2); // 推特
    private final int value;

    private EShare(int v) {
        this.value = v;
    }

    public String toString() {
        return this.value + "";
    }

    public static EShare get(int v) {
        String str = String.valueOf(v);
        return get(str);
    }

    public static EShare get(String str) {
        for (EShare e : values()) {
            if (e.toString().equals(str)) {
                return e;
            }
        }
        return null;
    }

    public static void main(String[] args) {
        EShare eShare = EShare.get(0);
    }
}
```

## 二

``` java
enum UserEnum {
	HTTP_200(200, "请求成功"),HTTP_500(500,"请求失败");
	private Integer code;
	private String name;

	UserEnum(Integer code, String name) {
		this.code = code;
		this.name = name;
	}

	public Integer getCode() {
		return code;
	}

	public void setCode(Integer code) {
		this.code = code;
	}

	public String getName() {
		return name;
	}

	public void setName(String name) {
		this.name = name;
	}

}

public class TestEnum {
	public static void main(String[] args) {
		System.out.println(UserEnum.HTTP_500.getCode());
	}

}

```
